# 项目使用Eslint

# AIM

  • 通过eslint规范代码
  • 保存自动格式化
  • vscode 快捷键格式化按照eslint规范进行

# HOW

  • cnpm i eslint --save-dev 来让固定的版本升级到更高的版本 避免refuse delete
  • npm 更新package-lock.json
  • 安装插件 ESLint Prettier Vetur
  • 配置ESLint规则
  • 选择Prettier来作为格式化的默认插件
  • 进行setting配置
{
    "eslint.autoFixOnSave": true,
    "prettier.eslintIntegration": true, //让prettier使用eslint的代码格式进行校验
    "prettier.semi": false, //去掉代码结尾的分号
    "prettier.trailingComma": "none", // 去掉代码结尾的逗号
    "prettier.singleQuote": true, //使用带引号替代双引号
    "javascript.format.insertSpaceBeforeFunctionParenthesis": true, //让函数(名)和后面的括号之间加个空格
    "vetur.format.defaultFormatter.html": "js-beautify-html", //格式化.vue中html
    "vetur.format.defaultFormatter.js": "vscode-typescript", //让vue中的js按编辑器自带的ts格式进行格式化
    "vetur.format.defaultFormatterOptions": {
        "js-beautify-html": {
            "wrap_attributes": "force-aligned" //属性强制折行对齐
        }
    },
    "eslint.validate": [
        "javascript", // 用eslint的规则检测js文件
        {
            "language": "vue", // 检测vue文件
            "autoFix": true // 为vue文件开启保存自动修复的功能
        },
        "javascriptreact",
        "html",
        "vue"
    ],
    "editor.codeActionsOnSave": {
        "source.fixAll.eslint": true
    },
    "[vue]": {
        "editor.defaultFormatter": "esbenp.prettier-vscode"
    }
}

# 识别webpack别名

需要做配置指定路径

module.exports = {
  verbose: true,
  moduleNameMapper: {
    '^@/(.*)$': '<rootDir>/src/$1',
    '^vue$': 'vue/dist/vue.common.js'
  },
  moduleFileExtensions: ['js', 'vue'],
  transform: {
    '^.+\\.js$': '<rootDir>/node_modules/babel-jest',
    '.*\\.(vue)$': '<rootDir>/node_modules/vue-jest'
  },
  transformIgnorePatterns: ['<rootDir>/node_modules/(?!(@babel))']
}
